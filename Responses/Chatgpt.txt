# store/gameStore.ts

```ts
"use client";

import { create } from "zustand";

interface GameState {
  booted: boolean;
  introFinished: boolean;
  currentSector: number;
  corruption: number;
  stability: number;
  achievements: string[];
  setBooted: (v: boolean) => void;
  setIntroFinished: (v: boolean) => void;
  nextSector: () => void;
  updateCorruption: (v: number) => void;
}

export const useGameStore = create<GameState>((set) => ({
  booted: false,
  introFinished: false,
  currentSector: 1,
  corruption: 86,
  stability: 13,
  achievements: [],
  setBooted: (v) => set({ booted: v }),
  setIntroFinished: (v) => set({ introFinished: v }),
  nextSector: () =>
    set((state) => ({
      currentSector: state.currentSector + 1
    })),
  updateCorruption: (v) => set({ corruption: v })
}));
```

```ts
export const executeCommand = (command: string) => {
  const cmd = command.toLowerCase();

  switch (cmd) {
    case "scan":
      return "Scanning corruption layers...";
    case "repair":
      return "Repair sequence initialized...";
    case "decrypt":
      return "Decrypting archived memory...";
    case "trace":
      return "Tracing neural pathways...";
    case "override":
      return "Override accepted.";
    case "stabilize":
      return "System stability increased.";
    case "inject":
      return "Injecting human logic...";
    default:
      return "Unknown command.";
  }
};
```

# components/ui/FloatingParticles.tsx

```tsx
"use client";

import { useEffect, useRef } from "react";

export default function FloatingParticles() {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    const canvas = canvasRef.current!;
    const ctx = canvas.getContext("2d")!;

    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;

    const particles = Array.from({ length: 200 }).map(() => ({
      x: Math.random() * canvas.width,
      y: Math.random() * canvas.height,
      r: Math.random() * 2,
      s: Math.random() * 1
    }));

    const render = () => {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      particles.forEach((p) => {
        p.y += p.s;

        if (p.y > canvas.height) p.y = 0;

        ctx.beginPath();
        ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
        ctx.fillStyle = "#00fff7";
        ctx.fill();
      });

      requestAnimationFrame(render);
    };

    render();
  }, []);

  return (
    <canvas
      ref={canvasRef}
      className="absolute inset-0 opacity-40"
    />
  );
}
```

```tsx
"use client";

import { Canvas, useFrame } from "@react-three/fiber";
import { Points, PointMaterial } from "@react-three/drei";
import * as THREE from "three";
import { useRef } from "react";

function Stars() {
  const ref = useRef<THREE.Points>(null);

  const positions = new Float32Array(5000);

  for (let i = 0; i < 5000; i++) {
    positions[i] = (Math.random() - 0.5) * 100;
  }

  useFrame((state) => {
    if (!ref.current) return;
    ref.current.rotation.y = state.clock.elapsedTime * 0.03;
  });

  return (
    <Points ref={ref} positions={positions} stride={3}>
      <PointMaterial
        transparent
        color="#00fff7"
        size={0.08}
        sizeAttenuation
      />
    </Points>
  );
}

export default function CyberScene() {
  return (
    <Canvas camera={{ position: [0, 0, 10] }}>
      <ambientLight intensity={0.5} />
      <Stars />
    </Canvas>
  );
}
```

```ts
import { openDB } from "idb";

export const dbPromise = openDB("error-human-found", 1, {
  upgrade(db) {
    db.createObjectStore("saves");
  }
});

export const saveData = async (
  key: string,
  value: any
) => {
  const db = await dbPromise;
  await db.put("saves", value, key);
};
```

# hooks/useAutosave.ts

```ts
"use client";

import { useEffect } from "react";
import { useGameStore } from "@/store/gameStore";
import { saveData } from "@/lib/indexedDB";

export default function useAutosave() {
  const game = useGameStore();

  useEffect(() => {
    const interval = setInterval(() => {
      localStorage.setItem(
        "ehf-save",
        JSON.stringify(game)
      );

      saveData("autosave", game);
    }, 5000);

    return () => clearInterval(interval);
  }, [game]);
}
```
